    public void addButton(JButton button, String rutaArchivo, String id, int cant) {
        button.addActionListener(e -> {
            Book li = new Book();
            new AbrirInstancia().instancia(li);
            String rutaFinal = rutaArchivo + id + 1 + ".txt";
            abrirArchivo(li.textArea, rutaFinal);
            li.cargarLibro(rutaArchivo, id, cant);
            this.setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
            this.dispose();
            this.requestFocusInWindow();
        });
    }
    
    public void abrirArchivo(JTextPane pane, String rutaL) {
        try (InputStream input = getClass().getResourceAsStream("/" + rutaL)) {
            if (input == null) {
                System.err.println("No se encontr√≥ el archivo: " + rutaL);
                return;
            }
            try (BufferedReader reader = new BufferedReader(new InputStreamReader(input, StandardCharsets.UTF_8))) {
                String contenido = reader.lines().collect(Collectors.joining("\n"));
                pane.setText(contenido);
                pane.setCaretPosition(0);
            }
        } catch (IOException ex) {
            ex.printStackTrace();
        }
    }
